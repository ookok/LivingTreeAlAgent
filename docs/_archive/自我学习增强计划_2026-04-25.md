class CodeEvolutionEnv:
    """代码进化环境（强化学习）"""
    
    def __init__(self, evolution_engine):
        self.evolution_engine = evolution_engine
        self.action_space = [
            'modify_parameter',
            'switch_strategy',
            'add_test_case',
            'refactor_code',
            'rollback_last_change'
        ]
        self.state = self._get_initial_state()
        self.steps = 0
        self.max_steps = 100
    
    def step(self, action):
        """执行动作，返回 (next_state, reward, done, info)"""
        # 1. 执行动作
        result = self._execute_action(action)
        
        # 2. 观察新状态
        next_state = self._observe_state()
        
        # 3. 计算奖励
        reward = self._calculate_reward(result)
        
        # 4. 检查是否结束
        self.steps += 1
        done = self.steps >= self.max_steps or result['success'] == False
        
        return next_state, reward, done, {'result': result}
    
    def reset(self):
        """重置环境"""
        self.state = self._get_initial_state()
        self.steps = 0
        return self.state
    
    def _calculate_reward(self, result):
        """计算奖励（多目标）"""
        reward = 0.0
        
        # 代码质量提升
        reward += result.get('quality_improvement', 0) * 10.0
        
        # 测试通过率
        reward += result.get('test_pass_rate', 0) * 5.0
        
        # 性能提升
        reward += result.get('performance_improvement', 0) * 3.0
        
        # 惩罚：引入新 bug
        reward -= result.get('new_bugs', 0) * 20.0
        
        # 惩罚：过度修改
        reward -= result.get('modification_size', 0) * 0.1
        
        return reward

#### 4.1.2 智能体（Agent）
```python
# client/src/business/self_learning/reinforcement/rl_agent.py

import torch
import torch.nn as nn
import torch.optim as optim
from torch.distributions import Categorical

class PolicyNetwork(nn.Module):
    """策略网络（Actor）"""
    
    def __init__(self, state_dim, action_dim, hidden_dim=128):
        super(PolicyNetwork, self).__init__()
        self.fc1 = nn.Linear(state_dim, hidden_dim)
        self.fc2 = nn.Linear(hidden_dim, hidden_dim)
        self.fc3 = nn.Linear(hidden_dim, action_dim)
    
    def forward(self, state):
        x = torch.relu(self.fc1(state))
        x = torch.relu(self.fc2(x))
        action_probs = torch.softmax(self.fc3(x), dim=-1)
        return action_probs

class ValueNetwork(nn.Module):
    """价值网络（Critic）"""
    
    def __init__(self, state_dim, hidden_dim=128):
        super(ValueNetwork, self).__init__()
        self.fc1 = nn.Linear(state_dim, hidden_dim)
        self.fc2 = nn.Linear(hidden_dim, hidden_dim)
        self.fc3 = nn.Linear(hidden_dim, 1)
    
    def forward(self, state):
        x = torch.relu(self.fc1(state))
        x = torch.relu(self.fc2(x))
        value = self.fc3(x)
        return value

class RLAgent:
    """强化学习智能体（PPO算法）"""
    
    def __init__(self, state_dim, action_dim, lr=3e-4, gamma=0.99, eps_clip=0.2):
        self.policy_net = PolicyNetwork(state_dim, action_dim)
        self.value_net = ValueNetwork(state_dim)
        self.optimizer = optim.Adam(list(self.policy_net.parameters()) + 
                                      list(self.value_net.parameters()), lr=lr)
        self.gamma = gamma
        self.eps_clip = eps_clip
        self.memory = []
    
    def select_action(self, state):
        """选择动作"""
        state = torch.tensor(state, dtype=torch.float32)
        action_probs = self.policy_net(state)
        dist = Categorical(action_probs)
        action = dist.sample()
        return action.item(), dist.log_prob(action)
    
    def store_transition(self, transition):
        """存储转移"""
        self.memory.append(transition)
    
    def update(self):
        """更新策略（PPO）"""
        states = torch.tensor([t[0] for t in self.memory], dtype=torch.float32)
        actions = torch.tensor([t[1] for t in self.memory], dtype=torch.int64)
        log_probs_old = torch.tensor([t[2] for t in self.memory], dtype=torch.float32)
        rewards = torch.tensor([t[3] for t in self.memory], dtype=torch.float32)
        dones = torch.tensor([t[4] for t in self.memory], dtype=torch.float32)
        
        # 计算 GAE（Generalized Advantage Estimation）
        advantages = self._compute_gae(rewards, dones)
        returns = advantages + self.value_net(states).detach()
        
        # PPO Update
        for _ in range(10):  # K-epochs
            log_probs_new = []
            values_new = self.value_net(states).squeeze()
            
            for i, state in enumerate(states):
                action_prob = self.policy_net(state)
                dist = Categorical(action_prob)
                log_prob = dist.log_prob(actions[i])
                log_probs_new.append(log_prob)
            
            log_probs_new = torch.stack(log_probs_new)
            
            # Ratio
            ratios = torch.exp(log_probs_new - log_probs_old)
            
            # Clipped loss
            surr1 = ratios * advantages
            surr2 = torch.clamp(ratios, 1-self.eps_clip, 1+self.eps_clip) * advantages
            actor_loss = -torch.min(surr1, surr2).mean()
            
            # Value loss
            value_loss = nn.MSELoss()(values_new, returns)
            
            # Total loss
            loss = actor_loss + 0.5 * value_loss
            
            self.optimizer.zero_grad()
            loss.backward()
            self.optimizer.step()
        
        self.memory.clear()
    
    def _compute_gae(self, rewards, dones):
        """计算 GAE"""
        gae = 0
        advantages = []
        
        for i in reversed(range(len(rewards))):
            if dones[i]:
                gae = rewards[i]
            else:
                gae = rewards[i] + self.gamma * gae
            
            advantages.insert(0, gae)
        
        return torch.tensor(advantages, dtype=torch.float32)
```

```python
# client/src/business/self_learning/reinforcement/trainer.py

class RLTrainer:
    """强化学习训练器"""
    
    def __init__(self, env, agent, max_episodes=1000):
        self.env = env
        self.agent = agent
        self.max_episodes = max_episodes
        self.episode_rewards = []
    
    def train(self):
        """训练循环"""
        for episode in range(self.max_episodes):
            state = self.env.reset()
            episode_reward = 0
            done = False
            
            while not done:
                # 选择动作
                action, log_prob = self.agent.select_action(state)
                
                # 执行动作
                next_state, reward, done, info = self.env.step(action)
                
                # 存储转移
                self.agent.store_transition((state, action, log_prob, reward, done))
                
                state = next_state
                episode_reward += reward
            
            # 更新策略
            self.agent.update()
            
            self.episode_rewards.append(episode_reward)
            
            if episode % 10 == 0:
                avg_reward = sum(self.episode_rewards[-10:]) / 10
                print(f"Episode {episode}, Avg Reward: {avg_reward:.2f}")
        
        print("Training completed!")
```

# client/src/business/self_learning/knowledge_graph/kg_auto_builder.py

import ast
import os
from pathlib import Path

class ASTParser:
    """AST 解析器（自动构建知识图谱）"""
    
    def __init__(self, kg_builder):
        self.kg_builder = kg_builder
        self.current_file = None
    
    def parse_file(self, file_path):
        """解析单个文件"""
        self.current_file = file_path
        
        with open(file_path, 'r', encoding='utf-8') as f:
            code = f.read()
        
        try:
            tree = ast.parse(code)
            self.kg_builder.add_file_node(
                file_path, 
                language='python',
                size=len(code),
                last_modified=os.path.getmtime(file_path)
            )
            self._visit(tree)
        except SyntaxError as e:
            print(f"Syntax error in {file_path}: {e}")
    
    def parse_directory(self, dir_path):
        """解析整个目录"""
        for py_file in Path(dir_path).rglob("*.py"):
            self.parse_file(str(py_file))
    
    def _visit(self, node):
        """遍历 AST"""
        if isinstance(node, ast.ClassDef):
            self._process_class(node)
        elif isinstance(node, ast.FunctionDef) or isinstance(node, ast.AsyncFunctionDef):
            self._process_function(node)
        
        for child in ast.iter_child_nodes(node):
            self._visit(child)
    
    def _process_class(self, node):
        """处理类定义"""
        class_node = self.kg_builder.add_class_node(
            class_name=node.name,
            file_path=self.current_file,
            start_line=node.lineno,
            end_line=node.end_lineno
        )
        
        # 处理继承
        for base in node.bases:
            if isinstance(base, ast.Name):
                self.kg_builder.add_inheritance_relationship(node.name, base.id)
    
    def _process_function(self, node):
        """处理函数定义"""
        func_node = self.kg_builder.add_function_node(
            func_name=node.name,
            file_path=self.current_file,
            start_line=node.lineno,
            end_line=node.end_lineno
        )
        
        # 处理函数调用
        self._extract_function_calls(node, node.name)
    
    def _extract_function_calls(self, node, current_func):
        """提取函数调用"""
        for child in ast.iter_child_nodes(node):
            if isinstance(child, ast.Call):
                if isinstance(child.func, ast.Name):
                    self.kg_builder.add_call_relationship(current_func, child.func.id)
                elif isinstance(child.func, ast.Attribute):
                    # method call: obj.method()
                    pass
            else:
                self._extract_function_calls(child, current_func)

#### 4.3.1 领域适配（Domain Adaptation）
```python
# client/src/business/self_learning/transfer/domain_adapter.py

import torch
import torch.nn as nn
import torch.nn.functional as F

class DomainAdapter(nn.Module):
    """领域适配器（迁移学习）"""
    
    def __init__(self, input_dim, hidden_dim=256):
        super(DomainAdapter, self).__init__()
        self.feature_extractor = nn.Sequential(
            nn.Linear(input_dim, hidden_dim),
            nn.ReLU(),
            nn.Dropout(0.2),
            nn.Linear(hidden_dim, hidden_dim),
            nn.ReLU(),
            nn.Dropout(0.2)
        )
        
        # 领域分类器（对抗训练）
        self.domain_classifier = nn.Sequential(
            nn.Linear(hidden_dim, 128),
            nn.ReLU(),
            nn.Linear(128, 2)  # 2 domains: source & target
        )
        
        # 任务分类器
        self.task_classifier = nn.Sequential(
            nn.Linear(hidden_dim, 128),
            nn.ReLU(),
            nn.Linear(128, 10)  # 10 tasks
        )
    
    def forward(self, x, alpha=1.0):
        """前向传播"""
        features = self.feature_extractor(x)
        
        # 反向传播时反转梯度（对抗训练）
        reversed_features = GradientReversalLayer.apply(features, alpha)
        domain_output = self.domain_classifier(reversed_features)
        
        task_output = self.task_classifier(features)
        
        return task_output, domain_output, features

class GradientReversalLayer(torch.autograd.Function):
    """梯度反转层（Domain Adversarial Training）"""
    
    @staticmethod
    def forward(ctx, x, alpha):
        ctx.alpha = alpha
        return x.view_as(x)
    
    @staticmethod
    def backward(ctx, grad_output):
        output = grad_output.neg() * ctx.alpha
        return output, None

class TransferTrainer:
    """迁移学习训练器"""
    
    def __init__(self, model, source_loader, target_loader, lr=1e-4):
        self.model = model
        self.source_loader = source_loader
        self.target_loader = target_loader
        self.optimizer = torch.optim.Adam(model.parameters(), lr=lr)
        self.task_criterion = nn.CrossEntropyLoss()
        self.domain_criterion = nn.CrossEntropyLoss()
    
    def train(self, epochs=100):
        """训练循环"""
        for epoch in range(epochs):
            # 训练源领域
            self.model.train()
            for batch_idx, (data, task_target) in enumerate(self.source_loader):
                domain_target = torch.zeros(data.size(0), dtype=torch.long)  # source = 0
                
                task_output, domain_output, _ = self.model(data, alpha=self._get_alpha(epoch))
                task_loss = self.task_criterion(task_output, task_target)
                domain_loss = self.domain_criterion(domain_output, domain_target)
                
                loss = task_loss + domain_loss
                
                self.optimizer.zero_grad()
                loss.backward()
                self.optimizer.step()
            
            # 训练目标领域
            for batch_idx, (data, _) in enumerate(self.target_loader):
                domain_target = torch.ones(data.size(0), dtype=torch.long)  # target = 1
                
                _, domain_output, _ = self.model(data, alpha=self._get_alpha(epoch))
                domain_loss = self.domain_criterion(domain_output, domain_target)
                
                self.optimizer.zero_grad()
                domain_loss.backward()
                self.optimizer.step()
            
            if epoch % 10 == 0:
                print(f"Epoch {epoch} completed")
        
        print("Transfer learning completed!")
    
    def _get_alpha(self, epoch, max_epochs=100):
        """动态调整 alpha（DANN 策略）"""
        return 2.0 / (1.0 + np.exp(-10.0 * epoch / max_epochs)) - 1.0
```

class CodeEvolutionPreTrainedModel:
    """代码进化预训练模型"""
    
    def __init__(self, model_name="codebert-base"):
        # 加载预训练模型（CodeBERT/GraphCodeBERT）
        self.tokenizer = AutoTokenizer.from_pretrained(model_name)
        self.model = AutoModel.from_pretrained(model_name)
        
        # 冻结底层（只训练顶层）
        for param in self.model.parameters():
            param.requires_grad = False
        
        # 添加自定义顶层
        self.classifier = nn.Sequential(
            nn.Linear(768, 256),
            nn.ReLU(),
            nn.Dropout(0.2),
            nn.Linear(256, 5)  # 5 classes: refactor/optimize/fix/feature/test
        )
    
    def forward(self, code_snippet, labels=None):
        """前向传播"""
        inputs = self.tokenizer(code_snippet, return_tensors="pt", padding=True, truncation=True)
        outputs = self.model(**inputs)
        
        # 取 [CLS] token 的输出
        cls_output = outputs.last_hidden_state[:, 0, :]
        
        logits = self.classifier(cls_output)
        
        if labels is not None:
            loss = nn.CrossEntropyLoss()(logits, labels)
            return loss, logits
        
        return logits
    
    def fine_tune(self, target_data, epochs=10):
        """微调（迁移到目标任务）"""
        # 解冻顶层
        for param in self.classifier.parameters():
            param.requires_grad = True
        
        optimizer = torch.optim.Adam(self.classifier.parameters(), lr=1e-4)
        
        for epoch in range(epochs):
            for code, label in target_data:
                loss, _ = self.forward(code, label)
                
                optimizer.zero_grad()
                loss.backward()
                optimizer.step()
            
            print(f"Fine-tuning Epoch {epoch} completed")
